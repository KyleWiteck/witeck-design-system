import {
  Button,
  FormField,
  Input
} from '@KyleWiteck/witeck-design/components';
import { PropsTable } from '@site/src/components/PropsTable';

A wrapper component that provides consistent layout and accessibility features
for form inputs.

## Usage

```tsx
import { FormField, Input } from '@KyleWiteck/witeck-design/components';

<FormField label="Email Address" htmlFor="email" required>
  <Input id="email" />
</FormField>;
```

## Examples

### Basic Usage

```tsx live
<FormField
  label="Username"
  htmlFor="username"
  helperText="Choose a unique username"
>
  <Input id="username" />
</FormField>
```

### Required Field

```tsx live
<FormField
  label="Password"
  htmlFor="password"
  required
  helperText="Must be at least 8 characters"
>
  <Input id="password" type="password" />
</FormField>
```

### With Error State

```tsx live
<FormField
  label="Email"
  htmlFor="email"
  required
  errorMessage="Please enter a valid email address"
>
  <Input id="email" type="email" />
</FormField>
```

### With Right Side Content

```tsx live
<FormField
  label="Profile Picture"
  htmlFor="avatar"
  labelRightSideContent={
    <Button variant="link" size="sm">
      Clear
    </Button>
  }
>
  <Input id="avatar" type="file" />
</FormField>
```

### Disabled State

```tsx live
<FormField
  label="Account Type"
  htmlFor="account-type"
  disabled
  helperText="Contact admin to change"
>
  <Input disabled />
</FormField>
```

## Best Practices

- Use clear, descriptive labels
- Provide helper text for complex fields
- Display actionable error messages
- Maintain consistent spacing
- Associate labels with controls using htmlFor
- Consider mobile viewport spacing

## Accessibility

- Semantic HTML structure
- Proper label association
- Screen reader support
- Keyboard navigation
- Error message announcements
- Support for hidden labels
